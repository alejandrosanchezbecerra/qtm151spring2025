import numpy as np
import matplotlib.pyplot as plt

coord_x = np.array([1, 4, 0, -1, 20, 40])
coord_y = np.array([2, 2, 0, 10, 0, 1])

hypotenuse = np.sqrt(coord_x**2 + coord_y**2)

plt.hist(hypotenuse, bins=10, color='purple', edgecolor='black')

# Label the axes and title
plt.xlabel('Hypotenuse')
plt.ylabel('Frequency')
plt.title('Histogram of Hypotenuse')

# Display the plot
plt.show()

import pandas as pd

datasets = ["data_quiz/china_co2_emissions.csv", "data_quiz/germany_co2_emissions.csv"]
list_emissions = []


for dataset in datasets:
    df = pd.read_csv(dataset)
    
    
    mean_emissions = df["total_emissions"].mean()
    list_emissions.append(mean_emissions)

print(list_emissions)

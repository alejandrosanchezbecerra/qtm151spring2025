df = pd.read_csv("data_quiz/fifa23_players_basic.csv")

    if TotalStats > 85:
        return "Top Performer"
    else:
        return "Non Top-Performer"

df["player_classification"] = df["TotalStats"].apply(classify_player)

# Write your own code
import pandas as pd
import numpy as np

df = pd.read_csv("data_quiz/fifa23_players_basic.csv")
n = df.shape[0]
df["random_var"] = np.random.uniform(0, 1, n)


data0 = df.query("random_var < 0.4")
data1 = df.query("random_var >= 0.4")

data0.to_csv("data0.csv", index=False)
data1.to_csv("data1.csv", index=False)

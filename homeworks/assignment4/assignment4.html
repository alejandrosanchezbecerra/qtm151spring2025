import matplotlib.pyplot as plt

x = np.arange(1, 16)

vec_y2 = x**4 - 6*x**2 + 2*x

list_vectors = [vec_y1, vec_y2]

    plt.scatter(x, vec, label=f'${"y_1" if vec is vec_y1 else "y_2"}$', s=100)
